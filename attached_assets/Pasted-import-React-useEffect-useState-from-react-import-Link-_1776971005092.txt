function ProgressBar({ percent, colorClass }: { percent: number; colorClass: string }) {
  const clamped = Math.max(0, Math.min(100, percent));
  const widthPct = clamped < 0.001 && clamped > 0 ? 0.5 : clamped; // ensure visible sliver
  return (
    <div className="w-full h-2 bg-secondary/40 rounded-full overflow-hidden">
      <div
        className={`h-full ${colorClass} rounded-full transition-all duration-500`}
        style={{ width: `${widthPct}%` }}
      />
    </div>
  );
}

function Row({ label, value, bold }: { label: string; value: string; bold?: boolean }) {
  return (
    <div className="flex justify-between items-center text-xs">
      <span className="text-muted-foreground">{label}</span>
      <span className={`font-mono ${bold ? "font-bold text-foreground" : "text-foreground/80"}`}>{value}</span>
    </div>
  );
}

function formatZbxCompact(n: number): string {
  if (n >= 1_000_000) return `${(n / 1_000_000).toFixed(2)}M`;
  if (n >= 1_000) return `${(n / 1_000).toFixed(2)}K`;
  if (n >= 1) return n.toLocaleString(undefined, { maximumFractionDigits: 4 });
  return n.toFixed(6);
}

type ZflTokenLite = {
  address: string;
  name: string;
  symbol: string;
  totalSupplyMicro: number;
  burnedTotal: number;
  circulatingSupply: number;
  logoUrl?: string | null;
  brandingTier?: string | null;
  masterPoolAddress?: string;
  masterTokenReserve: number;
  masterZusdtReserve: number;
  subPoolCount?: number;
  totalVolumeZusdt?: number;
  totalSwaps?: number;
  holderCount?: number;
  description?: string | null;
  website?: string | null;
};

function ZflTopTokensSection() {
  const [tokens, setTokens] = useState<ZflTokenLite[] | null>(null);

  useEffect(() => {
    let alive = true;
    let timer: any;
    async function load() {
      try {
        const r = await fetch("/api/zfl/tokens");
        if (!r.ok) return;
        const j = await r.json();
        if (!alive) return;
        setTokens((j.tokens ?? []) as ZflTokenLite[]);
      } catch {/* silent */}
    }
    load();
    timer = setInterval(load, 8_000);
    return () => { alive = false; clearInterval(timer); };
  }, []);

  if (!tokens) return null;

  const ranked = [...tokens]
    .map((t) => {
      const tokRes = t.masterTokenReserve ?? 0;
      const zRes = t.masterZusdtReserve ?? 0;
      const price = tokRes > 0 ? zRes / tokRes : 0;
      const mcap = price * (t.circulatingSupply / 1_000_000);
      const liquidity = zRes / 1_000_000;
      const volume = t.totalVolumeZusdt ?? 0;
      return { ...t, price, mcap, liquidity, zRes, volume };
    })
    .sort((a, b) => {
      // Primary: trade volume. Tiebreak: market cap, then liquidity.
      if (b.volume !== a.volume) return b.volume - a.volume;
      if (b.mcap !== a.mcap) return b.mcap - a.mcap;
      return b.liquidity - a.liquidity;
    })
    .slice(0, 6);

  return (
    <div className="space-y-4">
      <div className="flex items-center justify-between flex-wrap gap-2">
        <h2 className="text-xl font-bold tracking-tight text-foreground flex items-center gap-2">
          <span className="text-fuchsia-400">🪙</span>
          ZFL Token Launchpad
          <span className="ml-2 text-[11px] px-2 py-0.5 rounded-full border border-fuchsia-500/40 bg-fuchsia-500/10 text-fuchsia-200 font-medium uppercase tracking-wider">
            New
          </span>
        </h2>
        <div className="flex items-center gap-2 text-xs">
          <span className="text-muted-foreground">
            {tokens.length} token{tokens.length === 1 ? "" : "s"} on chain
          </span>
          <Link href="/zfl" className="text-fuchsia-300 hover:text-fuchsia-200 underline-offset-2 hover:underline">
            View all →
          </Link>
        </div>
      </div>

      {ranked.length === 0 ? (
        <Card className="border-dashed border-fuchsia-500/30 bg-fuchsia-500/5">
          <CardContent className="p-6 text-center">
            <div className="text-4xl mb-2">🚀</div>
            <div className="text-sm text-foreground font-medium mb-1">
              No tokens launched yet
            </div>
            <div className="text-xs text-muted-foreground mb-3">
              Be the first to launch a token on Zebvix Fabric Layer.
            </div>
            <Link href="/zfl/launch" className="inline-flex items-center gap-1.5 px-3 py-1.5 rounded-md bg-fuchsia-600 hover:bg-fuchsia-500 text-white text-xs font-medium">
              Launch token
            </Link>
          </CardContent>
        </Card>
      ) : (
        <div className="grid grid-cols-1 sm:grid-cols-2 lg:grid-cols-3 gap-3">
          {ranked.map((t, idx) => {
            const live = t.zRes > 0;
            const rank = idx + 1;
            const rankStyle =
              rank === 1
                ? "bg-amber-500/20 text-amber-300 border-amber-500/40"
                : rank === 2
                ? "bg-zinc-400/20 text-zinc-200 border-zinc-400/40"
                : rank === 3
                ? "bg-orange-700/30 text-orange-300 border-orange-700/50"
                : "bg-secondary text-muted-foreground border-border";
            return (
              <Link
                key={t.address}
                href={`/zfl/token/${t.address}`}
                className="block rounded-lg border border-border bg-card hover:border-fuchsia-500/50 hover:bg-fuchsia-500/5 transition-colors p-3"
              >
                <div className="flex items-center gap-3">
                  <div className="relative">
                    {t.logoUrl ? (
                      <img src={t.logoUrl} alt="" className="w-10 h-10 rounded-lg object-cover bg-secondary" />
                    ) : (
                      <div className="w-10 h-10 rounded-lg bg-gradient-to-br from-indigo-500 to-fuchsia-600 grid place-items-center text-xs font-bold text-white">
                        {t.symbol.slice(0, 2)}
                      </div>
                    )}
                    <span
                      className={`absolute -top-1.5 -left-1.5 text-[10px] font-bold w-5 h-5 grid place-items-center rounded-full border ${rankStyle}`}
                      title={`Rank #${rank} by 24h volume`}
                    >
                      {rank}
                    </span>
                  </div>
                  <div className="flex-1 min-w-0">
                    <div className="flex items-center gap-2">
                      <div className="font-semibold truncate">{t.name}</div>
                      {live ? (
                        <span className="text-[9px] px-1.5 py-0.5 rounded bg-emerald-900/40 text-emerald-300 border border-emerald-700/40">LIVE</span>
                      ) : (
                        <span className="text-[9px] px-1.5 py-0.5 rounded bg-amber-900/40 text-amber-300 border border-amber-700/40">DRY</span>
                      )}
                    </div>
                    <div className="text-[11px] text-muted-foreground font-mono truncate">
                      {t.symbol} · {t.address.slice(0, 8)}…{t.address.slice(-4)}
                    </div>
                  </div>
                </div>
                <div className="mt-3 grid grid-cols-4 gap-2 text-xs">
                  <div>
                    <div className="text-[10px] uppercase tracking-wider text-muted-foreground">Price</div>
                    <div className="font-mono font-semibold tabular-nums text-emerald-300">
                      {t.price > 0 ? t.price.toFixed(4) : "—"}
                    </div>
                  </div>
                  <div>
                    <div className="text-[10px] uppercase tracking-wider text-muted-foreground">Mcap</div>
                    <div className="font-mono font-semibold tabular-nums">
                      {t.mcap > 0 ? formatZbxCompact(t.mcap) : "—"}
                    </div>
                  </div>
                  <div>
                    <div className="text-[10px] uppercase tracking-wider text-muted-foreground">Liq</div>
                    <div className="font-mono font-semibold tabular-nums text-fuchsia-300">
                      {t.liquidity > 0 ? formatZbxCompact(t.liquidity) : "—"}
                    </div>
                  </div>
                  <div>
                    <div className="text-[10px] uppercase tracking-wider text-muted-foreground">Vol</div>
                    <div className="font-mono font-semibold tabular-nums text-cyan-300">
                      {t.volume > 0 ? formatZbxCompact(t.volume) : "—"}
                    </div>
                  </div>
                </div>
                <div className="mt-2 flex items-center justify-between text-[10px] text-muted-foreground">
                  <span>{t.totalSwaps ?? 0} swap{t.totalSwaps === 1 ? "" : "s"}</span>
                  <span>{t.holderCount ?? 0} holder{t.holderCount === 1 ? "" : "s"}</span>
                </div>
              </Link>
            );
          })}
        </div>
      )}
    </div>
  );
}

function formatCountdown(days: number | undefined): string {
  if (!days || days <= 0) return "imminent";
  if (days < 1) return `${(days * 24).toFixed(1)} hours`;
  if (days < 60) return `${days.toFixed(1)} days`;
  if (days < 730) return `${(days / 30.44).toFixed(1)} months`;
  return `${(days / 365.25).toFixed(2)} years`;
}
